### 5.3 Factory Pattern (Recommended)

```csharp
public static class DatabaseFactory
{
    public static ITransactionalStore CreateTransactional(
        string path,
        StorageEngine engine = StorageEngine.BTree,
        TransactionMode mode = TransactionMode.Wal,
        IBlockEncryptor? encryptor = null)
    {
        IKeyValueStore store = engine switch
        {
            StorageEngine.BTree => new BTreeStore(path + ".db"),
            StorageEngine.Lsm => new LsmTreeStore(path),
            _ => throw new ArgumentException()
        };
        
        ITransactionJournal? journal = mode switch
        {
            TransactionMode.Wal => new WalJournal(path + ".wal", encryptor),
            TransactionMode.RollbackJournal => new RollbackJournal(path, encryptor),
            TransactionMode.None => null,
            _ => null
        };
        
        var lockManager = new LockManager(path);
        
        return new TransactionalStore(store, journal, lockManager);
    }
}

public enum StorageEngine { BTree, Lsm }
```
